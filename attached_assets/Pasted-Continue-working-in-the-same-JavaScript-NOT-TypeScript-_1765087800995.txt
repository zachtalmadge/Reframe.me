function ResultsGuidanceSection({ hasNarratives, hasLetter }) {
  if (!hasNarratives && !hasLetter) return null;

  const [activeTab, setActiveTab] = useState(
    hasNarratives ? "narratives" : "letter"
  );

  const showPills = hasNarratives && hasLetter;

  return (
    <section className="mt-10 border-t border-slate-800 pt-8">
      <div className="max-w-3xl mx-auto space-y-4">
        <h2 className="text-xl sm:text-2xl font-semibold text-foreground">
          How to use what you just got
        </h2>

        {showPills && (
          <div className="inline-flex rounded-full bg-slate-900 p-1 text-sm">
            <button
              type="button"
              onClick={() => setActiveTab("narratives")}
              className={`px-4 py-1.5 rounded-full transition ${
                activeTab === "narratives"
                  ? "bg-slate-100 text-slate-900"
                  : "text-slate-200"
              }`}
            >
              Using your narratives
            </button>
            <button
              type="button"
              onClick={() => setActiveTab("letter")}
              className={`px-4 py-1.5 rounded-full transition ${
                activeTab === "letter"
                  ? "bg-slate-100 text-slate-900"
                  : "text-slate-200"
              }`}
            >
              Using your letter
            </button>
          </div>
        )}

        {/* Narratives guidance */}
        {hasNarratives && (!showPills || activeTab === "narratives") && (
          <div className="space-y-2 text-sm text-muted-foreground">
            {/* Placeholder guidance copy to be refined later */}
            <p>Practice reading one narrative out loud...</p>
          </div>
        )}

        {/* Letter guidance */}
        {hasLetter && (!showPills || activeTab === "letter") && (
          <div className="space-y-2 text-sm text-muted-foreground">
            {/* Placeholder guidance copy to be refined later */}
            <p>Read your letter carefully for accuracy...</p>
          </div>
        )}
      </div>
    </section>
  );
}

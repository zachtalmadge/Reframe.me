function MultiStepForm() {
  const [currentStepIndex, setCurrentStepIndex] = useState(0); // Step 0 = Step 1, etc.
  const [hasSeenEmotionalMessage, setHasSeenEmotionalMessage] = useState(false);

  return (
    <FormStepLayout
      currentStepIndex={currentStepIndex}
      hasSeenEmotionalMessage={hasSeenEmotionalMessage}
      setHasSeenEmotionalMessage={setHasSeenEmotionalMessage}
      // ...other props
    >
      {/* step content */}
    </FormStepLayout>
  );

function FormStepLayout({
  children,
  onBack,
  onNext,
  isNextDisabled,
  currentStepIndex,
  hasSeenEmotionalMessage,
  setHasSeenEmotionalMessage,
}) {
  const [showEmotionalMessage, setShowEmotionalMessage] = useState(
    hasSeenEmotionalMessage
  );

  useEffect(() => {
    let timeoutId;

    // Treat stepIndex 0 as "Step 1"
    if (currentStepIndex === 0 && !hasSeenEmotionalMessage) {
      timeoutId = setTimeout(() => {
        setShowEmotionalMessage(true);
        setHasSeenEmotionalMessage(true);
      }, 5000);
    } else if (hasSeenEmotionalMessage) {
      setShowEmotionalMessage(true);
    }

    return () => {
      if (timeoutId) clearTimeout(timeoutId);
    };
  }, [currentStepIndex, hasSeenEmotionalMessage, setHasSeenEmotionalMessage]);

  return (
    <form className="space-y-6">
      <div>{children}</div>

      <div className="flex items-center justify-between pt-4 border-t border-border">
        <button type="button" onClick={onBack} className="...">
          Back
        </button>
        <button type="submit" disabled={isNextDisabled} className="...">
          Continue
        </button>
      </div>

      <p
        className={`mt-3 text-xs text-center transition-opacity duration-300 text-orange-400 ${
          showEmotionalMessage ? "opacity-100" : "opacity-0"
        }`}
      >
        Talking about your past and your record can feel heavy. It’s okay to pause, take a
        break, or talk this through with someone you trust.
      </p>
    </form>
  );
}

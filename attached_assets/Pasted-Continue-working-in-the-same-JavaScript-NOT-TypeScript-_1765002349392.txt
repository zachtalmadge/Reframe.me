function Hero() {
  const [mounted, setMounted] = useState(false);

  useEffect(() => {
    const prefersReduced =
      window.matchMedia?.("(prefers-reduced-motion: reduce)").matches;
    if (!prefersReduced) {
      const id = requestAnimationFrame(() => setMounted(true));
      return () => cancelAnimationFrame(id);
    } else {
      setMounted(true);
    }
  }, []);

  return (
    <section className="px-4 py-12">
      <div className="relative max-w-3xl mx-auto overflow-hidden rounded-3xl">
        <div
          className={cn(
            "absolute inset-0 bg-gradient-to-r from-slate-900/80 to-slate-800/70 transition-transform transition-opacity duration-400",
            mounted ? "opacity-100 translate-y-0" : "opacity-0 translate-y-2"
          )}
          aria-hidden="true"
        />
        <div className="relative p-6 space-y-4">
          {/* existing hero text + CTA */}
          <div className="mt-4 text-xs text-left text-muted-foreground">
            <div>Before: “I freeze when employers ask about my record.”</div>
            <div className="mt-1 font-medium text-emerald-100">
              After: “Now I have language that explains my past and what I’ve done since.”
            </div>
          </div>
        </div>
      </div>
    </section>
  );
}
